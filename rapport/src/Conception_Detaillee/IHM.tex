   \subsubsection{Fonctions TAD}
     \paragraph{Console}
       \begin{algorithme}
         \fonction
         {console}
         {}
         {IHM}
         {ihm : IHM}
         {
           \affecter{ihm.afficherPlateau}{\pointeur{affichagePlateauConsole}}
           \affecter{ihm.afficherText}{\pointeur{affichageTexteConsole}}
           \retourner{ihm}
         }
       \end{algorithme}

      \paragraph{Tournoi}

      \paragraph{AffichagePlateauConsole}
        \begin{algorithme}
          \fonction
          {affichagePlateauConsole}
          {plateau : Plateau}
          {hauteau, largeur : \naturel , position : Position, cellule : Cellule, couleur : Couleur}
          {ihm : IHM}
          {
            \affecter{hauteau}{obtenirHauteur(plateau)}
            \affecter{largeur}{obtenirLargeur(plateau)}

            \pour{i}{'a'}{'a'+largeur-1}{1}
            {
              \ecrire{i}
            }
            \ecrire{Retour chariot}
            \pour{row}{0}{hauteur-1}{1}
            {
              \ecrire{row+1}
              \pour{col}{0}{largeur-1}{1}
              {
                \affecter{position}{obtenirHauteur(plateau)}
                \affecter{celulle}{obtenirCellule(plateau,position)}
                \affecter{couleur}{obtenirHauteur(cellule)}
                \ecrire{couleurVersChar(couleur)}
              }
              \ecrire{Retour chariot}
            }
          }
        \end{algorithme}

      \paragraph{afficherTexte}
