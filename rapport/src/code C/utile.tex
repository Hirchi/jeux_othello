#include "utile.h"
#include <ctype.h>
#include <stdlib.h>


unsigned int nbDigit(unsigned int nombre)
{
  unsigned int nombreDigit = 0;
  while((int)nombre) {
      nombre /= 10;
      nombreDigit++;
  }
  return nombreDigit;
}

unsigned int prochainNombreDansChaine(char* string)
{
  int i = 0;
  unsigned int nombre = 0;
  while (string[i] != '\0' && isdigit(string[i])) {
    nombre*= 10;
    nombre+= string[i] - '0';
    i++;
  }
  return nombre;
}
